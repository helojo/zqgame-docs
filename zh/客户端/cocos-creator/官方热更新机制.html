package/hot-update/main.js

(function () {
    if (typeof window.jsb === 'object') {
        var hotUpdateSearchPaths = localStorage.getItem('HotUpdateSearchPaths');
        if (hotUpdateSearchPaths) {
            var paths = JSON.parse(hotUpdateSearchPaths);
            jsb.fileUtils.setSearchPaths(paths);

            var fileList = [];
            var storagePath = paths[0] || '';
            var tempPath = storagePath + '_temp/';
            var baseOffset = tempPath.length;

            if (jsb.fileUtils.isDirectoryExist(tempPath) && !jsb.fileUtils.isFileExist(tempPath + 'project.manifest.temp')) {
                jsb.fileUtils.listFilesRecursively(tempPath, fileList);
                fileList.forEach(srcPath => {
                    var relativePath = srcPath.substr(baseOffset);
                    var dstPath = storagePath + relativePath;

                    if (srcPath[srcPath.length] == '/') {
                        cc.fileUtils.createDirectory(dstPath)
                    }
                    else {
                        if (cc.fileUtils.isFileExist(dstPath)) {
                            cc.fileUtils.removeFile(dstPath)
                        }
                        cc.fileUtils.renameFile(srcPath, dstPath);
                    }
                })
                cc.fileUtils.removeDirectory(tempPath);
            }
        }
    }
})();

大厅和子游戏整体更新模式

只在启动app时检测

  1. 大厅version检测和manifest文件版本比对更新
  2. 游戏列表gameList.json文件更新检测
  3. 根据gameList文件和本地gameList缓存检测子游戏bundle版本变化获取有版本更新的子游戏
  4. 根据有版本更新的子游戏进行游戏config.json文件版本比对更新

大厅和子游戏独立更新模式

分两个阶段做热更新检测

  1. 启动app时
1. 大厅version检测和manifest文件版本比对更新
2. 游戏列表gameList.json文件更新检测

  1. 进入子游戏时

    1. 根据gameList文件和本地gameList缓存检测子游戏bundle版本变化获取有版本更新的子游戏
    2. 根据有版本更新的子游戏进行游戏config.json文件版本比对更新
